18PUPRADAR_TEST(IntelHex_parses_one_data_record) {
19 // Address 0x00B8, 3 bytes [0x02, 0x00, 0xB8]
20 // Sum check: 03 + 00 + B8 + 00 + 02 + 00 + B8 = 0x175 → low byte 0x75 → checksum = 0x100-0x75 = 0x8B
21 // Build it programmatically:
22 auto sum = [](std::initializer_list<int> v) {
23 int s = 0; for (int b : v) s += b; return static_cast<std::uint8_t>((-s) & 0xFF);
24 };
25 int len = 3, ahi = 0x00, alo = 0xB8, type = 0x00;
26 int b0 = 0x02, b1 = 0x00, b2 = 0xB8;
27 auto cs = sum({len, ahi, alo, type, b0, b1, b2});
28 char line[64];
29 std::snprintf(line, sizeof(line), ":%02X%02X%02X%02X%02X%02X%02X%02X\n",
30 len, ahi, alo, type, b0, b1, b2, cs);
31 std::string text = line;
32 text += ":00000001FF\n";
33 std::istringstream s(text);
34 auto recs = parseIntelHex(s);
35 ASSERT_EQ(recs.size(), 1u);
36 ASSERT_EQ(recs[0].address, 0x00B8u);
37 ASSERT_EQ(recs[0].data.size(), 3u);
38 ASSERT_EQ(recs[0].data[0], 0x02u);
39 ASSERT_EQ(recs[0].data[1], 0x00u);
40 ASSERT_EQ(recs[0].data[2], 0xB8u);
41}
