11// Minimal JSON string escape — handles the few characters that can show up
12// in our metadata (paths with backslashes, ISO timestamps).
13std::string jsonEscape(const std::string& s) {
14 std::string out;
15 out.reserve(s.size() + 2);
16 for (char c : s) {
17 switch (c) {
18 case '"': out += "\\\""; break;
19 case '\\': out += "\\\\"; break;
20 case '\b': out += "\\b"; break;
21 case '\f': out += "\\f"; break;
22 case '\n': out += "\\n"; break;
23 case '\r': out += "\\r"; break;
24 case '\t': out += "\\t"; break;
25 default:
26 if (static_cast<unsigned char>(c) < 0x20) {
27 char buf[8];
28 std::snprintf(buf, sizeof(buf), "\\u%04x",
29 static_cast<unsigned>(c));
30 out += buf;
31 } else {
32 out += c;
33 }
34 }
35 }
36 return out;
37}

41std::string serializeMetadataJson(const CaptureMetadata& md) {
42 std::ostringstream o;
43 o << std::setprecision(15);
44 o << "{\n";
45 o << " \"format\": \"pupradar_capture/v1\",\n";
46 o << " \"timestamp_utc\": \"" << jsonEscape(md.timestamp_utc) << "\",\n";
47 o << " \"firmware_path\": \"" << jsonEscape(md.firmware_path) << "\",\n";
48 o << " \"f_low_hz\": " << md.f_low_hz << ",\n";
49 o << " \"f_high_hz\": " << md.f_high_hz << ",\n";
50 o << " \"bandwidth_hz\": " << (md.f_high_hz - md.f_low_hz) << ",\n";
51 o << " \"fs_hz\": " << md.fs_hz << ",\n";
52 o << " \"sweep_time_s\": " << md.sweep_time_s << ",\n";
53 o << " \"samples_per_sweep_per_rx\": " << md.samples_per_sweep_per_rx << ",\n";
54 o << " \"num_tx\": " << md.num_tx << ",\n";
55 o << " \"num_rx\": " << md.num_rx << ",\n";
56 o << " \"tx_mask\": " << static_cast<int>(md.tx_mask) << ",\n";
57 o << " \"rx_mask\": " << static_cast<int>(md.rx_mask) << ",\n";
58 o << " \"requested_duration_s\": " << md.requested_duration_s << ",\n";
59 o << " \"actual_duration_s\": " << md.actual_duration_s << ",\n";
60 o << " \"bytes_captured\": " << md.bytes_captured << ",\n";
61 o << " \"reads_completed\": " << md.reads_completed << ",\n";
62 o << " \"sample_dtype\": \"int16_interleaved_iq\",\n";
63 o << " \"endian\": \"little\",\n";
64 o << " \"board_info_words_hex\": [";
65 for (std::size_t i = 0; i < md.board_info_words.size(); ++i) {
66 if (i > 0) o << ", ";
67 o << "\"" << std::hex << std::uppercase << std::setw(4)
68 << std::setfill('0') << md.board_info_words[i] << "\"";
69 o << std::dec << std::nouppercase << std::setfill(' ');
70 }
71 o << "]\n";
72 o << "}\n";
73 return o.str();
74}
