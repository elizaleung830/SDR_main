1// pupradar_capture — capture-only MVP for the PUP EN24C T2R4 radar on Linux.
2//
3// Single-binary deployment: parses an Intel HEX firmware blob, downloads it
4// to the FX2LP via the standard Cypress 0xA0 loader, configures sawtooth FMCW,
5// captures raw IQ for N seconds, writes <out>.bin + <out>.json.
6//
7// Usage:
8// pupradar_capture --duration 5 --out /tmp/cap [--firmware /path/to.hex]
9// [--post-fw-vid 0x04B4 --post-fw-pid 0x8613]
10// [--fc-low 24.0e9 --fc-high 24.25e9]
11// [--sweep-time 2 --samp-num 1 --tx 1 --rx 1]
12// [--reenum-timeout-ms 5000 --bulk-timeout-ms 2000]
13//
14// Defaults: 24.0..24.25 GHz, 1 ms sweep, max samples, Tx1+Rx1.
15// On first hardware test, run `lsusb` after firmware load to discover the
16// post-firmware VID/PID and pass them via --post-fw-vid/--post-fw-pid.
17

56class Args {
57public:
58 explicit Args(int argc, char** argv) {
59 for (int i = 1; i < argc; ++i) {
60 std::string a = argv[i];
61 if (a == "-h" || a == "--help") { help_ = true; continue; }
62 if (a.rfind("--", 0) != 0) {
63 throw std::runtime_error("Unexpected positional argument: " + a);
64 }
65 if (i + 1 >= argc) {
66 throw std::runtime_error("Missing value for " + a);
67 }
68 kv_[a.substr(2)] = argv[++i];
69 }
70 }
71 bool help() const { return help_; }
72 bool has(const std::string& k) const { return kv_.count(k) != 0; }
73 std::string get(const std::string& k, const std::string& def = "") const {
74 auto it = kv_.find(k); return it == kv_.end() ? def : it->second;
75 }
76 double getDouble(const std::string& k, double def) const {
77 return has(k) ? std::stod(get(k)) : def;
78 }
79 int getInt(const std::string& k, int def) const {
80 return has(k) ? std::stoi(get(k), nullptr, 0) : def;
81 }
82 unsigned int getUint(const std::string& k, unsigned int def) const {
83 return has(k) ? static_cast<unsigned int>(std::stoul(get(k), nullptr, 0)) : def;
84 }
85private:
86 bool help_ = false;
87 std::map<std::string, std::string> kv_;
88};

92int main(int argc, char** argv) try {
93 Args args(argc, argv);
94 if (args.help()) { printUsage(argv[0]); return 0; }

103 cfg.f_low_hz = args.getDouble("fc-low", 24.0e9);
104 cfg.f_high_hz = args.getDouble("fc-high", 24.25e9);
105 cfg.sweep_time_idx = args.getInt ("sweep-time", 2);
106 cfg.samp_num_idx = args.getInt ("samp-num", 1);
107 cfg.tx_mask = static_cast<std::uint8_t>(args.getInt("tx", 1));
108 cfg.rx_mask = static_cast<std::uint8_t>(args.getInt("rx", 1));
109 cfg.bulk_timeout_ms = args.getUint ("bulk-timeout-ms", 2000);
110
111 const std::string out_base = args.get("out");
112 const std::string firmware = args.get("firmware", "firmware/SDR_USB_FW.hex");
113 const unsigned int reenum_ms = args.getUint("reenum-timeout-ms", 5000);
114
115 std::vector<pupradar::IUsbBackend::VidPid> post_fw;
116 post_fw.push_back({
117 static_cast<std::uint16_t>(args.getInt("post-fw-vid", 0x04B4)),
118 static_cast<std::uint16_t>(args.getInt("post-fw-pid", 0x8613))
119 });
120 if (args.has("post-fw-vid2") && args.has("post-fw-pid2")) {
121 post_fw.push_back({
122 static_cast<std::uint16_t>(args.getInt("post-fw-vid2", 0)),
123 static_cast<std::uint16_t>(args.getInt("post-fw-pid2", 0))
124 });
125 }
126
127 std::cout << "[pupradar] firmware: " << firmware << "\n";
128 std::cout << "[pupradar] out: " << out_base << ".bin / .json\n";
129 std::cout << "[pupradar] capture: " << cfg.duration_s << " s, "
130 << cfg.f_low_hz/1e9 << ".." << cfg.f_high_hz/1e9 << " GHz, "
131 << "sweep_idx=" << cfg.sweep_time_idx
132 << " samp_idx=" << cfg.samp_num_idx
133 << " tx_mask=0x" << std::hex << +cfg.tx_mask
134 << " rx_mask=0x" << +cfg.rx_mask << std::dec << "\n";
135
136 pupradar::UsbDevice usb;
137 pupradar::RadarSession session(usb, firmware);
138
139 std::cout << "[pupradar] initializing (open → claim → firmware → reopen)…\n";
140 session.initialize(post_fw, reenum_ms);
141
142 std::printf("[pupradar] board info (uint16 words, hex): ");
143 for (auto w : session.lastBoardInfo()) std::printf("%04X ", w);
144 std::printf("\n");
145
146 const pupradar::BoardInfo info = pupradar::decodeBoardInfo(session.lastBoardInfo());
147 if (info.valid) {
148 std::printf("[pupradar] model: %s\n", info.model_name);
149 std::printf("[pupradar] freq_band: %u num_tx: %u num_rx: %u\n",
150 info.freq_band, info.num_tx, info.num_rx);
151 std::printf("[pupradar] antenna_type: %u version: %u\n",
152 info.antenna_type, info.version);
153 } else {
154 std::printf("[pupradar] warning: FA05 signature not found in board info\n");
155 }
156
157 std::cout << "[pupradar] configuring sawtooth FMCW…\n";
158 session.configure(cfg);
159
160 pupradar::IqWriter writer(out_base);
161 auto sink = [&writer](const std::uint8_t* d, std::size_t n) {
162 writer.write(d, n);
163 };
164
165 std::cout << "[pupradar] capturing for " << cfg.duration_s << " s…\n";
166 auto md = session.captureIq(cfg, sink);
167 writer.finalize(md);
168
169 std::cout << "[pupradar] done: " << md.bytes_captured << " bytes in "
170 << md.reads_completed << " reads ("
171 << md.actual_duration_s << " s).\n";
172 std::cout << "[pupradar] wrote: " << writer.binPath()
173 << " + " << writer.jsonPath() << "\n";
174
175 session.shutdown();
176 return 0;
177} catch (const std::exception& e) {
178 std::cerr << "[pupradar] error: " << e.what() << "\n";
179 return 1;
