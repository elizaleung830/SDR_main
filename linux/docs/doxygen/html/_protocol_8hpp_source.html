51/// Build a single 512-byte bulk-OUT packet that the MCU will interpret.
52/// The MCU reads only the first uint16 (little-endian); we still send a full
53/// max-packet because the hardware MEX did, and short writes have historically
54/// caused FX2LP framing trouble on some hosts.
55std::array<std::uint8_t, kBulkOutPacketBytes>
56makeCommandPacket(std::uint8_t opcode, std::uint8_t parameter);

58/// Convenience overload: build the board-info request packet.
59inline std::array<std::uint8_t, kBulkOutPacketBytes>
60boardInfoPacket() { return makeCommandPacket(opcode::kBoardInfoRequest, 0x00); }

62/// Sawtooth FMCW PLL register pack — given low/high frequency in Hz and

66/// Math: BGT24 mixer ratio (F_PLL = F_Tx / 16) with a 50 MHz reference,
