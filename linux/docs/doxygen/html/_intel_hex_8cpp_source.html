11std::uint8_t hexNibble(char c) {
12 if (c >= '0' && c <= '9') return static_cast<std::uint8_t>(c - '0');
13 if (c >= 'A' && c <= 'F') return static_cast<std::uint8_t>(c - 'A' + 10);
14 if (c >= 'a' && c <= 'f') return static_cast<std::uint8_t>(c - 'a' + 10);
15 throw std::runtime_error("Intel HEX: non-hex character");
16}

18std::uint8_t hexByte(const std::string& s, std::size_t i) {
19 if (i + 1 >= s.size()) {
20 throw std::runtime_error("Intel HEX: truncated record");
21 }
22 return static_cast<std::uint8_t>((hexNibble(s[i]) << 4) | hexNibble(s[i + 1]));
23}

27std::vector<HexRecord> parseIntelHex(std::istream& input) {
28 std::vector<HexRecord> records;
29 std::string line;
30 bool saw_eof = false;
31 std::size_t line_no = 0;
32
33 while (std::getline(input, line)) {
34 ++line_no;
35 // Trim trailing CR/LF/whitespace
36 while (!line.empty() &&
37 (line.back() == '\r' || line.back() == '\n' ||
38 line.back() == ' ' || line.back() == '\t')) {
39 line.pop_back();
40 }
41 if (line.empty()) continue;
42
43 if (line[0] != ':') {
44 std::ostringstream oss;
45 oss << "Intel HEX line " << line_no << ": missing ':' prefix";
46 throw std::runtime_error(oss.str());
47 }
48 // Minimum: ':' + len(2) + addr(4) + type(2) + checksum(2) = 11 chars
49 if (line.size() < 11 || (line.size() % 2) == 0) {
50 std::ostringstream oss;
51 oss << "Intel HEX line " << line_no << ": bad length " << line.size();
52 throw std::runtime_error(oss.str());
53 }
54
55 const std::uint8_t byte_count = hexByte(line, 1);
56 const std::uint16_t address = static_cast<std::uint16_t>(
57 (hexByte(line, 3) << 8) | hexByte(line, 5));
58 const std::uint8_t rec_type = hexByte(line, 7);
59
60 // Expected line length = 1 + 2 + 4 + 2 + byte_count*2 + 2
61 const std::size_t expected = 11 + static_cast<std::size_t>(byte_count) * 2;
62 if (line.size() != expected) {
63 std::ostringstream oss;
64 oss << "Intel HEX line " << line_no
65 << ": expected " << expected << " chars, got " << line.size();
66 throw std::runtime_error(oss.str());
67 }
68
69 // Sum bytes (incl. checksum) — must be 0 mod 256
70 unsigned sum = 0;
71 for (std::size_t i = 1; i + 1 < line.size(); i += 2) {
72 sum += hexByte(line, i);
73 }
74 if ((sum & 0xFFu) != 0) {
75 std::ostringstream oss;
76 oss << "Intel HEX line " << line_no << ": checksum failure";
77 throw std::runtime_error(oss.str());
78 }
79
80 switch (rec_type) {
81 case 0x00: { // Data
82 HexRecord rec;
83 rec.address = address;
84 rec.data.reserve(byte_count);
85 for (std::uint8_t k = 0; k < byte_count; ++k) {
86 rec.data.push_back(hexByte(line, 9 + 2 * k));
87 }
88 records.push_back(std::move(rec));
89 break;
90 }
91 case 0x01: // EOF
92 saw_eof = true;
93 break;
94 case 0x02: case 0x03: case 0x04: case 0x05:
95 // Extended-address / start-segment records — irrelevant for FX2LP 8051
96 break;
97 default: {
98 std::ostringstream oss;
99 oss << "Intel HEX line " << line_no
100 << ": unknown record type 0x" << std::hex << +rec_type;
101 throw std::runtime_error(oss.str());
102 }
103 }
104
105 if (saw_eof) break;
106 }
107
108 if (!saw_eof) {
109 throw std::runtime_error("Intel HEX: missing EOF record (type 01)");
110 }
111 return records;
112}
