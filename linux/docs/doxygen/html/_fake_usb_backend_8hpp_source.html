57class FakeUsbBackend final : public IUsbBackend {
58public:
59 std::vector<UsbEvent> events;

64 void open(std::uint16_t vid, std::uint16_t pid) override {

66 events.push_back(e);
67 is_open = true;
68 }
69 void close() override {
70 if (!is_open) return;
71 UsbEvent e; e.kind = UsbEvent::Kind::Close; events.push_back(e);
72 is_open = false;
73 }

75
76 void claimInterface(int n) override {
77 UsbEvent e; e.kind = UsbEvent::Kind::ClaimInterface; e.interface_no = n;
78 events.push_back(e);
79 }
80 void setAltSetting(int iface, int alt) override {
81 UsbEvent e; e.kind = UsbEvent::Kind::SetAlt;
82 e.interface_no = iface; e.alt_setting = alt;
83 events.push_back(e);
84 }
85
86 std::size_t controlTransfer(std::uint8_t bm, std::uint8_t br,
87 std::uint16_t wv, std::uint16_t wi,
88 std::uint8_t* data, std::size_t length,
89 unsigned int /*timeout*/) override {
90 UsbEvent e;
91 e.bm_request_type = bm; e.b_request = br;
92 e.w_value = wv; e.w_index = wi; e.requested_len = length;
93 // Direction bit is MSB of bm
94 if (bm & 0x80) {
95 // IN
96 e.kind = UsbEvent::Kind::ControlIn;
97 if (control_in_responses.empty()) {
98 throw std::runtime_error("FakeUsbBackend: no scripted control IN response");
99 }
100 auto resp = std::move(control_in_responses.front());
101 control_in_responses.pop_front();
102 std::size_t n = std::min(resp.size(), length);
103 std::memcpy(data, resp.data(), n);
104 events.push_back(e);
105 return n;
106 }
107 // OUT
108 e.kind = UsbEvent::Kind::ControlOut;
109 e.payload.assign(data, data + length);
110 events.push_back(e);
111 return length;
112 }
113
114 std::size_t bulkWrite(std::uint8_t ep, const std::uint8_t* data,
115 std::size_t length, unsigned int) override {
116 UsbEvent e; e.kind = UsbEvent::Kind::BulkOut; e.endpoint = ep;
117 e.payload.assign(data, data + length);
118 events.push_back(e);
119 return length;
120 }
121
122 std::size_t bulkRead(std::uint8_t ep, std::uint8_t* data,
123 std::size_t length, unsigned int) override {
124 UsbEvent e; e.kind = UsbEvent::Kind::BulkIn; e.endpoint = ep;
125 e.requested_len = length;
126 events.push_back(e);
127 if (bulk_in_responses.empty()) {
128 // Default: return requested zero-fill so tests that just want the
129 // session to "make progress" don't have to script every read.
130 std::memset(data, 0, length);
131 return length;
132 }
133 auto resp = std::move(bulk_in_responses.front());
134 bulk_in_responses.pop_front();
135 std::size_t n = std::min(resp.size(), length);
136 std::memcpy(data, resp.data(), n);
137 return n;
138 }
139
140 void reopenAfterRenumeration(const VidPid* cands, std::size_t n,
141 unsigned int) override {
142 UsbEvent e; e.kind = UsbEvent::Kind::ReopenAfterRenum;
143 if (n > 0) { e.vid = cands[0].vid; e.pid = cands[0].pid; }
144 events.push_back(e);
145 is_open = true;
146 }
147
148 // Helpers
149 std::size_t countOf(UsbEvent::Kind k) const {
150 std::size_t c = 0;
151 for (const auto& e : events) if (e.kind == k) ++c;
152 return c;
153 }
154
155 std::vector<UsbEvent> only(UsbEvent::Kind k) const {
156 std::vector<UsbEvent> out;
157 for (const auto& e : events) if (e.kind == k) out.push_back(e);
158 return out;
159 }
160};
