65/**
66 * @brief Packs an opcode/parameter pair into a 512-byte bulk-OUT packet and sends it to EP2.
67 *
68 * The MCU interprets only the first 16-bit little-endian word (opcode | param << 8), but the
69 * full 512-byte max-packet must be sent to avoid FX2LP framing issues observed on some hosts.
70 * @param opcode Command opcode byte (see pupradar::opcode namespace in Protocol.hpp).
71 * @param parameter Command parameter byte (semantics depend on opcode).
72 * @throws std::runtime_error if fewer bytes than the full 512-byte packet are transferred.
73 */
74void RadarSession::sendCommand(std::uint8_t opcode, std::uint8_t parameter) {
75 auto pkt = makeCommandPacket(opcode, parameter);
76 std::size_t n = usb_.bulkWrite(kEpOutBulk, pkt.data(), pkt.size(),
77 kBulkOutTimeoutMs);
78 if (n != pkt.size()) {
79 std::ostringstream oss;
80 oss << "sendCommand(0x" << std::hex << +opcode << ",0x" << +parameter
81 << "): short write " << std::dec << n << "/" << pkt.size();
82 throw std::runtime_error(oss.str());
83 }
84}

86/**
87 * @brief Downloads the Intel HEX firmware image into FX3 RAM via the Cypress 0xA0 loader protocol.
88 *
89 * Sequence mirrors usbdownload.cpp:
90 * 1. Read CPUCS register (0xE600), set bit 0 to hold the 8051 in reset.
91 * 2. Write each HEX data record to its target address via control transfers.
92 * 3. Clear CPUCS bit 0 to release reset — the 8051 begins executing firmware.
93 *
94 * After this returns the chip re-enumerates under a new VID/PID. Callers must invoke
95 * IUsbBackend::reopenAfterRenumeration() before issuing any further transfers.
96 * @throws std::runtime_error if the HEX file contains no data records or if any
97 * control transfer completes with fewer bytes than expected.
98 */
99void RadarSession::downloadFirmware() {
100 auto records = parseIntelHexFile(fw_path_);
101 if (records.empty()) {
102 throw std::runtime_error("Firmware HEX has no data records");
103 }
104
105 // Step 1: read CPUCS, set bit 0 (hold 8051 in reset), write back.
106 std::uint8_t cpucs = 0;
107 usb_.controlTransfer(kCtrlIn, kFwLoaderRequest, kCpucsRegister, 0,
108 &cpucs, 1, kCtrlTimeoutMs);
109 cpucs = static_cast<std::uint8_t>(cpucs | kCpucsReset);
110 usb_.controlTransfer(kCtrlOut, kFwLoaderRequest, kCpucsRegister, 0,
111 &cpucs, 1, kCtrlTimeoutMs);
112
113 // Step 2: write each data record into 8051 RAM.
114 for (const auto& rec : records) {
115 if (rec.data.empty()) continue;
116 std::size_t n = usb_.controlTransfer(
117 kCtrlOut, kFwLoaderRequest, rec.address, 0,
118 const_cast<std::uint8_t*>(rec.data.data()),
119 rec.data.size(), kCtrlTimeoutMs);
120 if (n != rec.data.size()) {
121 std::ostringstream oss;
122 oss << "Firmware download: short write at addr 0x"
123 << std::hex << rec.address << " (" << std::dec
124 << n << "/" << rec.data.size() << ")";
125 throw std::runtime_error(oss.str());
126 }
127 }
128
129 // Step 3: clear CPUCS bit 0 → 8051 starts running firmware.
130 cpucs = static_cast<std::uint8_t>(cpucs & ~kCpucsReset);
131 usb_.controlTransfer(kCtrlOut, kFwLoaderRequest, kCpucsRegister, 0,
132 &cpucs, 1, kCtrlTimeoutMs);
133}

135/**
136 * @brief Sends the 0xFA board-info request and parses the response into last_board_info_.
137 *
138 * Mirrors the GUI at line 2674: sends opcode kBoardInfoRequest, bulk-reads
139 * kBoardInfoReadBytes (512 + 2048) bytes from EP6, then extracts up to
140 * kBoardInfoExposedBytes starting at kBoardInfoUsefulOffsetBytes (2048),
141 * decoded as little-endian uint16 words.
142 *
143 * Hardware note: the device may return fewer than kBoardInfoReadBytes bytes.
144 * The observed minimum is ~2058 (useful_offset + 10), which covers all 5
145 * semantically meaningful words from the MATLAB GUI (lines 2675-2683).
146 * The decode is therefore bounded by however many bytes actually arrived,
147 * not by the nominal kBoardInfoExposedBytes.
148 *
149 * @throws std::runtime_error if the bulk read returns fewer than
150 * kBoardInfoUsefulOffsetBytes + 2 bytes (less than 1 uint16 word).
151 */
152void RadarSession::requestBoardInfo() {
153 sendCommand(opcode::kBoardInfoRequest, 0x00);
154 std::vector<std::uint8_t> buf(kBoardInfoReadBytes, 0);
155 std::size_t n = usb_.bulkRead(kEpInBulk, buf.data(), buf.size(),
156 /*timeout_ms*/ 2000);
157 if (n < kBoardInfoUsefulOffsetBytes + 2) {
158 std::ostringstream oss;
159 oss << "Board info: short read " << n
160 << " bytes (need at least " << (kBoardInfoUsefulOffsetBytes + 2) << ")";
161 throw std::runtime_error(oss.str());
162 }
163 // Decode however many complete uint16 words arrived, up to kBoardInfoExposedBytes.
164 const std::size_t avail_bytes = n - kBoardInfoUsefulOffsetBytes;
165 const std::size_t decode_bytes = (std::min(kBoardInfoExposedBytes, avail_bytes) / 2) * 2;
166 last_board_info_.clear();
167 last_board_info_.reserve(decode_bytes / 2);
168 for (std::size_t i = 0; i < decode_bytes; i += 2) {
169 std::uint8_t lo = buf[kBoardInfoUsefulOffsetBytes + i];
170 std::uint8_t hi = buf[kBoardInfoUsefulOffsetBytes + i + 1];
171 last_board_info_.push_back(
172 static_cast<std::uint16_t>((hi << 8) | lo));
173 }
174}

176/**
177 * @brief Full device bring-up: open pre-firmware device, download firmware, wait for
178 * re-enumeration, reclaim the interface, and fetch board info.
179 *
180 * Mirrors PUPradar_initiating() in the MATLAB GUI. Steps:
181 * 1. Open the unprogrammed FX2LP (VID 0x04B4 / PID 0x8613).
182 * 2. Claim interface 0, set alternate setting 1 (benign if the descriptor lacks it).
183 * 3. Download firmware via the Cypress 0xA0 loader (downloadFirmware()).
184 * 4. Wait for the chip to re-enumerate under one of post_fw_candidates.
185 * 5. Claim interface 0 and set alternate setting 1 again (now required for bulk EPs).
186 * 6. Request board info (requestBoardInfo()) — result available via lastBoardInfo().
187 *
188 * @param post_fw_candidates Non-empty list of (VID, PID) pairs the chip may present after
189 * the firmware boots. Tried in order until one opens or the timeout
190 * elapses.
191 * @param reenum_timeout_ms Maximum time in milliseconds to wait for re-enumeration
192 * (default: 5000 ms).
193 * @throws std::invalid_argument if post_fw_candidates is empty.
194 * @throws UsbError or std::runtime_error on any USB failure.
195 */
196void RadarSession::initialize(
197 const std::vector<IUsbBackend::VidPid>& post_fw_candidates,
198 unsigned int reenum_timeout_ms) {
199 if (post_fw_candidates.empty()) {
200 throw std::invalid_argument("Need at least one post-firmware VID/PID candidate");
201 }
202
203 // Pre-firmware open
204 usb_.open(kPreFirmwareVid, kPreFirmwarePid);
205 usb_.claimInterface(kInterfaceNumber);
206
207 // Set alt 1 — this matches PUPradar_initiating ordering. On the
208 // unprogrammed FX2LP this is benign because the default descriptor
209 // exposes a single interface 0 with one alt setting.
210 try {
211 usb_.setAltSetting(kInterfaceNumber, kAlternateSetting);
212 } catch (const UsbError&) {
213 // Some unprogrammed FX2LP descriptors do not have alt 1; ignore here,
214 // we'll set it again post-firmware where it matters.
215 }
216
217 downloadFirmware();
218
219 // Re-enumeration: chip detaches and re-appears with firmware-defined VID/PID.
220 usb_.reopenAfterRenumeration(post_fw_candidates.data(),
221 post_fw_candidates.size(),
222 reenum_timeout_ms);
223 usb_.claimInterface(kInterfaceNumber);
224 usb_.setAltSetting(kInterfaceNumber, kAlternateSetting);
225
226 requestBoardInfo();
227}

229/**
230 * @brief Pushes a full set of FMCW parameters to the radar over USB.
231 *
232 * Sends commands in the same order as the MATLAB GUI (SetActiveParameters →
233 * Send_Basic_Parameter → Send_PLL_Sawtooth): modulation, sweep time index, sampling
234 * number index, Tx mask, Rx mask, then all PLL register bytes for the sawtooth ramp.
235 * @param cfg Capture configuration describing the desired waveform. The MVP requires
236 * cfg.modulation == Modulation::Sawtooth; CW is deferred.
237 * @throws std::invalid_argument if cfg.modulation is not Modulation::Sawtooth.
238 * @throws std::runtime_error on any USB write failure (propagated from sendCommand).
239 */
240void RadarSession::configure(const CaptureConfig& cfg) {
241 if (cfg.modulation != Modulation::Sawtooth) {
242 throw std::invalid_argument("MVP supports Sawtooth only (CW deferred)");
243 }
244
245 // Order mirrors GUI: SetActiveParameters → Send_Basic_Parameter → Send_PLL_Sawtooth.
246 // Modulation
247 sendCommand(opcode::kModulation, static_cast<std::uint8_t>(cfg.modulation));
248 // Sweep time index
249 sendCommand(opcode::kSweepTimeIndex,
250 static_cast<std::uint8_t>(cfg.sweep_time_idx));
251 // Sampling number index
252 sendCommand(opcode::kSamplingNumIndex,
253 static_cast<std::uint8_t>(cfg.samp_num_idx));
254 // Tx mask
255 sendCommand(opcode::kTxMask, cfg.tx_mask);
256 // Rx mask
257 sendCommand(opcode::kRxMask, cfg.rx_mask);
258
259 // PLL register byte writes (sawtooth)
260 auto pll = buildSawtoothPllCommands(cfg.f_low_hz, cfg.f_high_hz,
261 cfg.sweep_time_idx);
262 for (const auto& c : pll) sendCommand(c.opcode, c.param);
263}

265/**
266 * @brief Bulk-reads raw IQ data from EP6 for cfg.duration_s seconds, forwarding each
267 * chunk to sink.
268 *
269 * Buffer sizing mirrors the GUI at line 2730: (kNumSweepsPerRead=64 + 40) sweeps worth of
270 * raw bytes, rounded up to a 512-byte boundary, plus 4096 bytes of padding. The capture
271 * loop runs until std::chrono::steady_clock exceeds the computed deadline; zero-length
272 * reads are silently skipped. No DSP is performed — this is a capture-only MVP.
273 *
274 * @param cfg Capture configuration used for buffer sizing, run duration, and bulk timeout.
275 * Must match the configuration previously sent via configure().
276 * @param sink Callback invoked with each non-empty bulk read chunk (raw pointer into an
277 * internal buffer, valid only for the duration of the call).
278 * @return CaptureMetadata populated with echoed config values, derived sample rate (fs_hz),
279 * sweep time, actual capture duration, byte and read counts, last board info words,
280 * firmware path, and a UTC ISO-8601 timestamp.
281 * @throws std::runtime_error on a USB error during bulk transfer (propagated from bulkRead).
282 */
283CaptureMetadata RadarSession::captureIq(const CaptureConfig& cfg,
284 const IqSink& sink) {
285 const int n_rx = rxCount(cfg.rx_mask);
286 const int n_tx = txCount(cfg.tx_mask);
287 const int samples_per_sweep_per_rx =
288 samplesPerSweep(cfg.sweep_time_idx, cfg.samp_num_idx, n_rx);
289 const double sweep_time_s = sweepTimeFromIndex(cfg.sweep_time_idx);
290 const double fs_hz = static_cast<double>(samples_per_sweep_per_rx) / sweep_time_s;
291
292 // Read size mirrors GUI line 2730: round to 512-byte boundary, plus 4096
293 // padding. NumSweeps batched to 64 to match the GUI's run loop, which is
294 // what the firmware's USB FIFO is sized for.
295 constexpr int kNumSweepsPerRead = 64;
296 // samplesPerSweep() returns the BASN value, which is already a uint16 count
297 // with I and Q interleaved (both channels). Multiplying by 2 for int16 byte
298 // width is correct; multiplying again for I,Q would double-count and produce
299 // a buffer 2× too large, causing libusb to wait beyond the FX3's burst size.
300 const std::size_t bytes_per_sweep =
301 static_cast<std::size_t>(samples_per_sweep_per_rx) *
302 2u /* bytes per uint16 */ *
303 static_cast<std::size_t>(n_rx) * static_cast<std::size_t>(n_tx);
304 const std::size_t raw_bytes_per_read =
305 static_cast<std::size_t>(kNumSweepsPerRead + 40) * bytes_per_sweep;
306 // Round UP to 512-byte boundary, then add 4096
307 const std::size_t bytes_per_read =
308 ((raw_bytes_per_read + 511u) / 512u) * 512u + 4096u;
309
310 std::vector<std::uint8_t> buf(bytes_per_read, 0);
311
312 using clock = std::chrono::steady_clock;
313 const auto t_start = clock::now();
314 const auto t_end = t_start +
315 std::chrono::milliseconds(static_cast<long>(cfg.duration_s * 1000.0));
316
317 std::size_t total_bytes = 0;
318 std::size_t reads_done = 0;
319
320 while (clock::now() < t_end) {
321 std::size_t got = usb_.bulkRead(kEpInBulk, buf.data(), buf.size(),
322 cfg.bulk_timeout_ms);
323 if (got == 0) continue;
324 sink(buf.data(), got);
325 total_bytes += got;
326 ++reads_done;
327 }
328 const auto t_done = clock::now();
329
330 CaptureMetadata md;
331 md.f_low_hz = cfg.f_low_hz;
332 md.f_high_hz = cfg.f_high_hz;
333 md.fs_hz = fs_hz;
334 md.sweep_time_s = sweep_time_s;
335 md.samples_per_sweep_per_rx = samples_per_sweep_per_rx;
336 md.num_tx = n_tx;
337 md.num_rx = n_rx;
338 md.tx_mask = cfg.tx_mask;
339 md.rx_mask = cfg.rx_mask;
340 md.requested_duration_s = cfg.duration_s;
341 md.actual_duration_s = std::chrono::duration<double>(t_done - t_start).count();
342 md.bytes_captured = total_bytes;
343 md.reads_completed = reads_done;
344 md.board_info_words = last_board_info_;
345 md.firmware_path = fw_path_;
346 md.timestamp_utc = isoTimestampUtcNow();
347 return md;
348}
