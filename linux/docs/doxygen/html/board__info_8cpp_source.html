1// pupradar_board_info — minimal USB connectivity diagnostic for the PUP EN24C T2R4 radar.
2//
3// Two modes:
4// Default: open pre-firmware device (VID 0x04B4/PID 0x8613), download firmware,
5// wait for re-enumeration, then request and print board info.
6// --skip-firmware: device already has firmware in RAM (USB cable was NOT unplugged since
7// the last firmware load); connect directly to the post-firmware VID/PID.
8//
9// Usage:
10// pupradar_board_info [--firmware <path>] [--post-fw-vid <VID>] [--post-fw-pid <PID>]

39// Direct board-info query used when firmware is already resident.
40// Opens the post-firmware device, claims interface 0 / alt 1, sends opcode 0xFA,
41// reads back the response, and decodes the useful region as little-endian uint16 words.
42std::vector<std::uint16_t> queryBoardInfoDirect(
43 std::uint16_t post_vid, std::uint16_t post_pid)
44{
45 // GUI line 2672: DataLength = 512 + 2048
46 constexpr std::size_t kReadBytes = 512 + 2048;
47 // GUI line 2674: PUPradarBoardInfo(1025:1100) — word 1025 (1-based) = byte 2048
48 constexpr std::size_t kUsefulOffset = 2048;
49 // GUI reads 76 words but only words 0-4 have meaning (FA05, FreqBand, Tx/Rx, Version).
50 // 32 bytes = 16 words gives comfortable headroom over those 5 semantic words.
51 constexpr std::size_t kExposedBytes = 32;
52
53 pupradar::UsbDevice usb;
54 usb.open(post_vid, post_pid);
55 usb.claimInterface(pupradar::kInterfaceNumber);
56 usb.setAltSetting(pupradar::kInterfaceNumber, pupradar::kAlternateSetting);
57
58 auto pkt = pupradar::makeCommandPacket(pupradar::opcode::kBoardInfoRequest, 0x00);
59 usb.bulkWrite(pupradar::kEpOutBulk, pkt.data(), pkt.size(), /*timeout_ms*/ 1000);
60
61 std::vector<std::uint8_t> buf(kReadBytes, 0);
62 std::size_t n = usb.bulkRead(pupradar::kEpInBulk, buf.data(), buf.size(),
63 /*timeout_ms*/ 2000);
64 if (n < kUsefulOffset + 2) {
65 throw std::runtime_error("Board info short read: got " + std::to_string(n) +
66 " bytes, need at least " +
67 std::to_string(kUsefulOffset + 2));
68 }
69
70 // Decode however many complete uint16 words arrived, up to kExposedBytes.
71 const std::size_t avail_bytes = n - kUsefulOffset;
72 const std::size_t decode_bytes = (std::min(kExposedBytes, avail_bytes) / 2) * 2;
73 std::vector<std::uint16_t> words;
74 words.reserve(decode_bytes / 2);
75 for (std::size_t i = 0; i < decode_bytes; i += 2) {
76 std::uint8_t lo = buf[kUsefulOffset + i];
77 std::uint8_t hi = buf[kUsefulOffset + i + 1];
78 words.push_back(static_cast<std::uint16_t>((hi << 8) | lo));
79 }
80
81 usb.close();
82 return words;
83}

87int main(int argc, char** argv) try {

97 if (i + 1 >= argc) throw std::runtime_error("Missing value for " + a);
98 return argv[++i];
99 };
100 if (a == "--firmware") firmware = next();
101 else if (a == "--post-fw-vid") post_vid = static_cast<std::uint16_t>(std::stoul(next(), nullptr, 0));
102 else if (a == "--post-fw-pid") post_pid = static_cast<std::uint16_t>(std::stoul(next(), nullptr, 0));
103 else if (a == "--reenum-timeout-ms") reenum_ms = static_cast<unsigned int>(std::stoul(next()));
104 else if (a == "--skip-firmware") skip_fw = true;
105 else if (a == "-h" || a == "--help") { printHelp(argv[0]); return 0; }
106 else throw std::runtime_error("Unknown argument: " + a);
107 }
108
109 std::vector<std::uint16_t> words;
110
111 if (skip_fw) {
112 std::printf("[board_info] mode: direct (skip firmware download)\n");
113 std::printf("[board_info] opening post-firmware device 0x%04X:0x%04X …\n",
114 post_vid, post_pid);
115 words = queryBoardInfoDirect(post_vid, post_pid);
116 } else {
117 std::printf("[board_info] mode: full init (firmware download + re-enumerate)\n");
118 std::printf("[board_info] firmware: %s\n", firmware.c_str());
119 std::printf("[board_info] opening pre-firmware device (VID 0x%04X / PID 0x%04X) …\n",
120 pupradar::kPreFirmwareVid, pupradar::kPreFirmwarePid);
121
122 pupradar::UsbDevice usb;
123 pupradar::RadarSession session(usb, firmware);
124
125 std::vector<pupradar::IUsbBackend::VidPid> post_fw = {{ post_vid, post_pid }};
126 session.initialize(post_fw, reenum_ms);
127 words = session.lastBoardInfo();
128 session.shutdown();
129 }
130
131 std::printf("[board_info] board info (%zu uint16 words, hex):\n ", words.size());
132 for (auto w : words) std::printf("%04X ", w);
133 std::printf("\n");
134
135 const pupradar::BoardInfo info = pupradar::decodeBoardInfo(words);
136 if (info.valid) {
137 std::printf("[board_info] signature: FA05 (OK)\n");
138 std::printf("[board_info] freq_band: %u\n", info.freq_band);
139 std::printf("[board_info] num_tx: %u\n", info.num_tx);
140 std::printf("[board_info] num_rx: %u\n", info.num_rx);
141 std::printf("[board_info] antenna_type: %u\n", info.antenna_type);
142 std::printf("[board_info] version: %u\n", info.version);
143 std::printf("[board_info] model code: %u\n", info.modelcode);
144 std::printf("[board_info] model: %s\n", info.model_name);
145 } else {
146 std::printf("[board_info] signature mismatch (word[0] = %04X, expected FA05)\n",
147 words.empty() ? 0u : static_cast<unsigned>(words[0]));
148 }
149 std::printf("[board_info] OK\n");
150 return 0;
151
152} catch (const std::exception& e) {
153 std::fprintf(stderr, "[board_info] error: %s\n", e.what());
154 return 1;
155}
