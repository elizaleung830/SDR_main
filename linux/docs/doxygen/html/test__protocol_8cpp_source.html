102PUPRADAR_TEST(Protocol_pll_start_n_at_24GHz_low_band) {
103 // F_start = 24e9/16 = 1.5e9; Start_N = 1.5e9/50e6 = 30.0 → integer 30, frac 0
104 auto cmds = buildSawtoothPllCommands(24.0e9, 24.25e9, 2);
105 // PLL Reg 03 L, M, H = LSB, mid, MSB of 30 = 0x1E, 0x00, 0x00
106 // Find them
107 auto get = [&](std::uint8_t op) {
108 for (auto& c : cmds) if (c.opcode == op) return static_cast<int>(c.param);
109 throw std::runtime_error("opcode not in cmds");
110 };
111 ASSERT_EQ(get(opcode::kPll03L), 30);
112 ASSERT_EQ(get(opcode::kPll03M), 0);
113 ASSERT_EQ(get(opcode::kPll03H), 0);
114 // PLL Reg 04 (frac) = round(0 * 2^24) = 0
115 ASSERT_EQ(get(opcode::kPll04L), 0);
