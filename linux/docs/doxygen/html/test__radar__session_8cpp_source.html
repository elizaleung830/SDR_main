17// Generate a tiny one-record + EOF Intel HEX file at a temp path.
18std::string writeTinyFirmware() {
19 // Build a valid 1-byte data record at addr 0x0100, value 0x42
20 int len = 1, ahi = 0x01, alo = 0x00, type = 0x00, b0 = 0x42;
21 int s = len + ahi + alo + type + b0;
22 int cs = (-s) & 0xFF;
23 char line[64];
24 std::snprintf(line, sizeof(line), ":%02X%02X%02X%02X%02X%02X\n",
25 len, ahi, alo, type, b0, cs);
26 std::string path = "tiny_fw.hex";
27 std::ofstream f(path, std::ios::trunc);
28 f << line << ":00000001FF\n";
29 f.close();
30 return path;
31}

33// Build a board-info response: 2048 bytes of header zeros, then 0xFA 0x05
34// little-endian, then 30 more bytes of dummy info.
35std::vector<std::uint8_t> makeBoardInfoResponse() {
36 std::vector<std::uint8_t> r(2560, 0);
37 r[2048] = 0x05; // little-endian 0xFA05
38 r[2049] = 0xFA;
39 r[2050] = 0xF0; // FrequencyBand byte
40 r[2051] = 0x00;
41 return r;
42}

45PUPRADAR_TEST(Session_initialize_sequence_is_correct) {
46 FakeUsbBackend usb;

51 // 1 control OUT (clear reset)
